# 链表

## 单向链表

```c++
// 链表节点结构体
struct ListNode {
    int val;                // 数据域
    ListNode* next;         // 指针域，指向下一个节点
    
    // 构造函数（易错点：必须初始化next为nullptr）
    ListNode(int x) : val(x), next(nullptr) {}
};
```

```c++
class LinkedList {
private:
    ListNode* head;  // 头指针
    int size;        // 链表长度

public:
    // 构造函数
    LinkedList() : head(nullptr), size(0) {}
    
    // 析构函数（易错点：必须释放所有节点）
    ~LinkedList() {
        clear();
    }
    
    // 清空链表
    void clear() {
        ListNode* current = head;
        while (current != nullptr) {
            ListNode* next = current->next;  // ⚠️ 先保存下一个节点
            delete current;                   // 再释放当前节点
            current = next;                   // 移动到下一个
        }
        head = nullptr;
        size = 0;
    }
    
    // 头插法
    void insertAtHead(int val) {
        ListNode* newNode = new ListNode(val);
        newNode->next = head;  // ⚠️ 新节点的next先指向原头节点
        head = newNode;        // 再更新头指针
        size++;
    }
    
    // 尾插法
    void insertAtTail(int val) {
        ListNode* newNode = new ListNode(val);
        if (head == nullptr) {  // ⚠️ 空链表特殊情况
            head = newNode;
        } else {
            ListNode* current = head;
            while (current->next != nullptr) {  // 找到最后一个节点
                current = current->next;
            }
            current->next = newNode;
        }
        size++;
    }
    
    // 在指定位置插入
    bool insertAtPosition(int position, int val) {
        if (position < 0 || position > size) {  // ⚠️ 边界检查
            return false;
        }
        
        if (position == 0) {
            insertAtHead(val);
            return true;
        }
        
        ListNode* current = head;
        for (int i = 0; i < position - 1; i++) {
            current = current->next;
        }
        
        ListNode* newNode = new ListNode(val);
        newNode->next = current->next;  // ⚠️ 顺序很重要！
        current->next = newNode;
        size++;
        return true;
    }
    
    // 删除头节点
    bool deleteAtHead() {
        if (head == nullptr) {  // ⚠️ 空链表检查
            return false;
        }
        
        ListNode* temp = head;
        head = head->next;  // ⚠️ 先更新头指针
        delete temp;        // 再释放原头节点
        size--;
        return true;
    }
    
    // 删除指定值节点
    bool deleteByValue(int val) {
        if (head == nullptr) {
            return false;
        }
        
        // 删除头节点的情况
        if (head->val == val) {
            return deleteAtHead();
        }
        
        ListNode* current = head;
        // ⚠️ 找到要删除节点的前一个节点
        while (current->next != nullptr && current->next->val != val) {
            current = current->next;
        }
        
        if (current->next == nullptr) {  // 没找到
            return false;
        }
        
        ListNode* temp = current->next;
        current->next = current->next->next;  // ⚠️ 跳过要删除的节点
        delete temp;
        size--;
        return true;
    }
    
    // 查找节点
    ListNode* find(int val) {
        ListNode* current = head;
        while (current != nullptr) {
            if (current->val == val) {
                return current;
            }
            current = current->next;
        }
        return nullptr;
    }
    
    // 遍历打印
    void print() {
        ListNode* current = head;
        while (current != nullptr) {
            cout << current->val << " -> ";
            current = current->next;
        }
        cout << "nullptr" << endl;
    }
    
    // 获取长度
    int getSize() {
        return size;
    }
};
```

## 双向链表

```c++
// 双向链表节点
struct DoublyNode {
    int val;
    DoublyNode* prev;  // 指向前一个节点
    DoublyNode* next;  // 指向后一个节点
    
    DoublyNode(int x) : val(x), prev(nullptr), next(nullptr) {}
};

class DoublyLinkedList {
private:
    DoublyNode* head;
    DoublyNode* tail;  // 尾指针，方便尾插
    int size;

public:
    DoublyLinkedList() : head(nullptr), tail(nullptr), size(0) {}
    
    ~DoublyLinkedList() {
        clear();
    }
    
    void clear() {
        DoublyNode* current = head;
        while (current != nullptr) {
            DoublyNode* next = current->next;
            delete current;
            current = next;
        }
        head = tail = nullptr;
        size = 0;
    }
    
    // 头插法（双向链表）
    void insertAtHead(int val) {
        DoublyNode* newNode = new DoublyNode(val);
        if (head == nullptr) {
            head = tail = newNode;
        } else {
            newNode->next = head;
            head->prev = newNode;  // ⚠️ 双向链表需要维护prev指针
            head = newNode;
        }
        size++;
    }
    
    // 尾插法（双向链表更高效）
    void insertAtTail(int val) {
        DoublyNode* newNode = new DoublyNode(val);
        if (tail == nullptr) {
            head = tail = newNode;
        } else {
            newNode->prev = tail;
            tail->next = newNode;  // ⚠️ 维护双向指针
            tail = newNode;
        }
        size++;
    }
    
    // 删除节点
    bool deleteByValue(int val) {
        DoublyNode* current = head;
        while (current != nullptr && current->val != val) {
            current = current->next;
        }
        
        if (current == nullptr) return false;
        
        // 更新前驱节点的next
        if (current->prev != nullptr) {
            current->prev->next = current->next;
        } else {
            head = current->next;  // 删除的是头节点
        }
        
        // 更新后继节点的prev
        if (current->next != nullptr) {
            current->next->prev = current->prev;
        } else {
            tail = current->prev;  // 删除的是尾节点
        }
        
        delete current;
        size--;
        return true;
    }
};
```

## 易错点

### 1、内存泄漏

```c++
// ❌ 错误：没有析构函数，节点从未释放
class LinkedList {
    ListNode* head;
public:
    LinkedList() : head(nullptr) {}
    // 缺少析构函数！程序结束时内存泄漏
};

// ✅ 正确：析构函数释放所有节点
~LinkedList() {
    clear();
}
```

### 2、删除顺序

```c++
// ❌ 错误：先delete再访问next，导致野指针
ListNode* temp = head;
delete temp;
head = temp->next;  // ❌ temp已释放，访问非法内存！

// ✅ 正确：先保存next，再delete
ListNode* temp = head;
head = head->next;  // 先更新指针
delete temp;        // 再释放内存
```

### 3、插入顺序

```c++
// ❌ 错误：先修改current->next，丢失后续节点
ListNode* newNode = new ListNode(val);
current->next = newNode;      // ❌ 原链表从此断开
newNode->next = current->next; // 实际指向自己！

// ✅ 正确：先连接新节点到后续，再连接前驱到新节点
ListNode* newNode = new ListNode(val);
newNode->next = current->next;  // 新节点指向原后续
current->next = newNode;        // 前驱指向新节点
```

### 4、空指针解引用

```c++
// ❌ 错误：未检查空指针
ListNode* current = head;
while (current->next != nullptr) {  // 如果head是nullptr，直接崩溃！
    current = current->next;
}

// ✅ 正确：先检查当前节点
ListNode* current = head;
while (current != nullptr && current->next != nullptr) {
    current = current->next;
}
```

### 5、头节点特殊处理

```c++
// ❌ 错误：删除时未考虑头节点
bool deleteByValue(int val) {
    ListNode* current = head;
    while (current->next->val != val) {  // 如果要删除的是头节点，这里会出错
        current = current->next;
    }
    // ...
}

// ✅ 正确：单独处理头节点
bool deleteByValue(int val) {
    if (head == nullptr) return false;
    if (head->val == val) {
        return deleteAtHead();  // 单独处理头节点
    }
    // 处理其他节点...
}
```

## 使用虚拟头节点

```c++
#include <iostream>
using namespace std;

// 链表节点
struct ListNode {
    int val;
    ListNode* next;
    ListNode(int x) : val(x), next(nullptr) {}
};

// 使用虚拟头节点的链表类
class LinkedList {
private:
    ListNode* dummy;  // 虚拟头节点
    int size;

public:
    // 构造函数
    LinkedList() {
        dummy = new ListNode(0);  // 创建虚拟头节点
        size = 0;
    }
    
    // 析构函数
    ~LinkedList() {
        clear();
        delete dummy;  // 释放虚拟头节点
    }
    
    // 清空链表
    void clear() {
        ListNode* current = dummy->next;  // 从真实头节点开始
        while (current != nullptr) {
            ListNode* next = current->next;
            delete current;
            current = next;
        }
        dummy->next = nullptr;
        size = 0;
    }
    
    // 头插法（统一处理）
    void insertAtHead(int val) {
        insertAfter(dummy, val);  // 在dummy后插入
    }
    
    // 尾插法
    void insertAtTail(int val) {
        ListNode* current = dummy;
        while (current->next != nullptr) {
            current = current->next;
        }
        insertAfter(current, val);
    }
    
    // 在指定节点后插入（核心方法）
    void insertAfter(ListNode* node, int val) {
        ListNode* newNode = new ListNode(val);
        newNode->next = node->next;
        node->next = newNode;
        size++;
    }
    
    // 在指定位置插入
    bool insertAtPosition(int position, int val) {
        if (position < 0 || position > size) return false;
        
        ListNode* current = dummy;  // 从dummy开始
        for (int i = 0; i < position; i++) {
            current = current->next;
        }
        insertAfter(current, val);
        return true;
    }
    
    // 删除指定值节点（统一处理）
    bool deleteByValue(int val) {
        ListNode* current = dummy;  // 从dummy开始
        
        // 找到要删除节点的前驱
        while (current->next != nullptr && current->next->val != val) {
            current = current->next;
        }
        
        if (current->next == nullptr) return false;
        
        ListNode* temp = current->next;
        current->next = current->next->next;
        delete temp;
        size--;
        return true;
    }
    
    // 删除指定位置节点
    bool deleteAtPosition(int position) {
        if (position < 0 || position >= size) return false;
        
        ListNode* current = dummy;  // 从dummy开始
        for (int i = 0; i < position; i++) {
            current = current->next;
        }
        
        ListNode* temp = current->next;
        current->next = current->next->next;
        delete temp;
        size--;
        return true;
    }
    
    // 获取真实头节点
    ListNode* getHead() {
        return dummy->next;
    }
    
    // 遍历打印
    void print() {
        ListNode* current = dummy->next;  // 从真实头节点开始
        while (current != nullptr) {
            cout << current->val << " -> ";
            current = current->next;
        }
        cout << "nullptr" << endl;
    }
    
    int getSize() {
        return size;
    }
};
```
